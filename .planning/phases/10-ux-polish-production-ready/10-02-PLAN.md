def get_message(key, lang='zh', **kwargs):
    """
    Get translated message by key.

    Args:
        key: Message key (e.g., 'error_sensitive')
        lang: Language code ('zh' or 'en'), defaults to 'zh'
        **kwargs: Format arguments for message templating

    Returns:
        Translated message string

    Example:
        >>> get_message('error_sensitive', lang='zh')
        '检测到敏感信息'

        >>> get_message('info_files_scanned', lang='en', count=10)
        'Files scanned: 10'
    """
    # Get language dictionary (fallback to Chinese if lang not found)
    lang_dict = MESSAGES.get(lang, MESSAGES['zh'])

    # Get message (fallback to key if message not found)
    message = lang_dict.get(key, key)

    # Apply template formatting if kwargs provided
    if kwargs:
        try:
            message = message.format(**kwargs)
        except KeyError:
            # If formatting fails, return unformatted message
            pass

    return message


def get_available_languages():
    """Return list of supported language codes."""
    return list(MESSAGES.keys())

def format_issue(issue, use_colors=True, lang='zh'):
    """Format a single issue for display with color and severity."""

    # Determine severity prefix and color
    severity = issue.get('severity', Severity.ERROR)

    if severity == Severity.ERROR:
        prefix = get_message('error_prefix', lang)
        color = Fore.RED if use_colors else ''
    else:  # WARNING
        prefix = get_message('warning_prefix', lang)
        color = Fore.YELLOW if use_colors else ''

    # Get issue type message
    issue_type = issue.get('type', 'UNKNOWN')
    message_key = issue.get('message_key', 'error_sensitive')
    message = get_message(message_key, lang)

    # Format issue details
    file_path = issue.get('file', 'unknown')
    line_number = issue.get('line', 0)

    # Build formatted string
    if use_colors:
        severity_str = f"{color}{prefix}{Style.RESET_ALL}"
    else:
        severity_str = prefix

    formatted = f"{severity_str} {issue_type}: {message}\n"
    formatted += f"  File: {file_path}:{line_number}\n"

    # Add content snippet if available
    if 'content' in issue:
        content = issue['content']
        if use_colors:
            formatted += f"  Content: {Fore.CYAN}{content}{Style.RESET_ALL}\n"
        else:
            formatted += f"  Content: {content}\n"

    # Add fix suggestion if available
    if 'suggestion' in issue:
        suggestion_key = issue['suggestion']
        suggestion = get_message(suggestion_key, lang)
        formatted += f"  Suggestion: {suggestion}\n"

    return formatted
```

def generate_report(issues, use_colors=True, lang='zh'):
    """Generate formatted report with color and language support."""

    if not issues:
        message = get_message('info_scan_complete', lang)
        if use_colors:
            print(f"{Fore.GREEN}✓{Style.RESET_ALL} {message}")
        else:
            print(f"✓ {message}")
        return

    # Print summary
    error_count = sum(1 for i in issues if i.get('severity') == Severity.ERROR)
    warning_count = sum(1 for i in issues if i.get('severity') == Severity.WARNING)

    if use_colors:
        print(f"\n{Fore.RED}Security Issues Found:{Style.RESET_ALL}")
    else:
        print("\nSecurity Issues Found:")

    print(f"  {get_message('info_errors', lang, count=error_count)}")
    print(f"  {get_message('info_warnings', lang, count=warning_count)}")
    print()

    # Print each issue
    for issue in issues:
        print(format_issue(issue, use_colors, lang))

    # Print blocking message if needed
    if should_block_commit(issues):
        block_msg = get_message('block_commit', lang)
        if use_colors:
            print(f"{Fore.RED}✗{Style.RESET_ALL} {block_msg}")
        else:
            print(f"✗ {block_msg}")

        print(get_message('block_fix_issues', lang))
        print(f"  {get_message('block_whitelist_help', lang)}")
```

<color_output>
## Color Output

The scanner uses colored output to improve readability.

**Default:** Colors enabled (auto-detected)

**Disable Colors:**

```bash
python scanner.py --no-color
```

**Color Scheme:**

- **ERROR** (red): Blocks commit, must be resolved
- **WARNING** (yellow): Shows warning (future feature)
- **INFO** (blue/green): Informational messages

**Terminal Support:**

Colors work automatically on:
- ✓ Windows CMD
- ✓ Windows PowerShell
- ✓ Git Bash
- ✓ Other terminals with ANSI support

**Graceful Degradation:**

When output is redirected to a file or terminal doesn't support colors:
- Colors are automatically disabled
- Plain text output (no ANSI codes)

```bash
# Redirect to file (no colors)
python scanner.py > scan_results.txt
```

</color_output>

<severity_levels>
## Severity Levels

Issues are classified by severity level:

**ERROR (blocks commit):**
- Sensitive information (API keys, passwords, tokens)
- Configuration files (.env, credentials.json)
- SSH/PGP private keys
- Internal information (private IPs, internal domains)

**WARNING (future feature):**
- Non-critical issues that don't block commit
- Currently not used, reserved for future extensions

**Behavior:**

- **ERROR** issues must be resolved or whitelisted before commit
- **WARNING** issues show a warning but allow commit (future)

**Output Format:**

```
[ERROR] Sensitive information detected
[WARNING] Cache file detected
```

</severity_levels>
