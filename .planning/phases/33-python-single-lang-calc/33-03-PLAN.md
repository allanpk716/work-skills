**Python-specific test helpers:**

```python
import json
import os
from pathlib import Path

def read_codepoint_log():
    """Read the latest codepoint log file."""
    log_dir = Path.home() / ".codepoint" / "python-calculator"
    if not log_dir.exists():
        return []
    log_files = sorted(log_dir.glob("cp-python-*.log"), key=lambda f: f.stat().st_mtime, reverse=True)
    if not log_files:
        return []
    entries = []
    for line in log_files[0].read_text(encoding='utf-8').splitlines():
        line = line.strip()
        if line.startswith('{'):
            try:
                entries.append(json.loads(line))
            except json.JSONDecodeError:
                continue
    return entries

def filter_by_point_id(entries, point_id):
    return [e for e in entries if e.get('name') == point_id]

def get_stack_string(entry):
    """Get stack info from Python codepoint entry (frames format)."""
    frames = entry.get('frames', [])
    if frames:
        return '\n'.join(f.get('func', '') for f in frames)
    return entry.get('stack', '')
```

def run_api_flow(expr="2+3*4"):
    from api.server import create_app
    from calculator.core import evaluate
    from history.store import HistoryStore

    store = HistoryStore()
    result, err = evaluate(expr, flow_id="flow-api-calculate")
    if not err:
        store.add(expr, result)
    return result, err

```python
def run_history_flow(expr="2+3*4"):
    from api.server import create_app
    from calculator.core import evaluate
    from history.store import HistoryStore

    store = HistoryStore()
    # First add via API
    result, err = evaluate(expr, flow_id="flow-api-calculate")
    store.add(expr, result)

    # Now query history and RECOMPUTE
    record = store.get(1)
    recomputed, _ = evaluate(record["expression"], flow_id="flow-history-query")
    return recomputed
```
