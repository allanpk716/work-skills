```python
def format_issue(issue, use_colors=True, lang='zh'):
    """Format a single issue for display with color and severity."""

    # Determine severity prefix and color
    severity = issue.get('severity', Severity.ERROR)

    if severity == Severity.ERROR:
        prefix = '[ERROR]' if lang == 'en' else '[错误]'
        color = Fore.RED if use_colors else ''
    else:  # WARNING
        prefix = '[WARNING]' if lang == 'en' else '[警告]'
        color = Fore.YELLOW if use_colors else ''

    # Format issue details
    issue_type = issue.get('type', 'UNKNOWN')
    file_path = issue.get('file', 'unknown')
    line_number = issue.get('line', 0)
    message = issue.get('message', 'Issue detected')

    # Build formatted string
    if use_colors:
        severity_str = f"{color}{prefix}{Style.RESET_ALL}"
    else:
        severity_str = prefix

    formatted = f"{severity_str} {issue_type}: {message}\n"
    formatted += f"  File: {file_path}:{line_number}\n"

    # Add content snippet if available
    if 'content' in issue:
        content = issue['content']
        if use_colors:
            formatted += f"  Content: {Fore.CYAN}{content}{Style.RESET_ALL}\n"
        else:
            formatted += f"  Content: {content}\n"

    # Add fix suggestion if available
    if 'suggestion' in issue:
        suggestion = issue['suggestion']
        formatted += f"  Suggestion: {suggestion}\n"

    return formatted
```

3. 添加 severity 过滤函数:
```python
def should_block_commit(issues):
    """Determine if commit should be blocked based on issues."""
    # Block commit if any ERROR severity issues found
    return any(
        issue.get('severity', Severity.ERROR) == Severity.ERROR
        for issue in issues
    )
```

4. 更新 `generate_report()` 函数签名:
```python
def generate_report(issues, use_colors=True, lang='zh'):
    """Generate formatted report with color support."""

    if not issues:
        if use_colors:
            print(f"{Fore.GREEN}✓{Style.RESET_ALL} Security scan passed. No issues found.")
        else:
            print("✓ Security scan passed. No issues found.")
        return

    # Print summary
    error_count = sum(1 for i in issues if i.get('severity') == Severity.ERROR)
    warning_count = sum(1 for i in issues if i.get('severity') == Severity.WARNING)

    if use_colors:
        print(f"\n{Fore.RED}Security Issues Found:{Style.RESET_ALL}")
    else:
        print("\nSecurity Issues Found:")

    print(f"  Errors: {error_count}")
    print(f"  Warnings: {warning_count}")
    print()

    # Print each issue
    for issue in issues:
        print(format_issue(issue, use_colors, lang))

    # Print blocking message if needed
    if should_block_commit(issues):
        if use_colors:
            print(f"{Fore.RED}✗{Style.RESET_ALL} Commit blocked due to security issues.")
        else:
            print("✗ Commit blocked due to security issues.")
        print("Please fix the issues above or add whitelist comments.")
```

**Verification:**
- format_issue() 支持 use_colors 和 lang 参数
- ERROR 级别使用红色输出
- WARNING 级别使用黄色输出
- should_block_commit() 函数存在且逻辑正确
- generate_report() 支持 use_colors 参数
</task>
