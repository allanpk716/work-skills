```python
"""Core calculator with shared pipeline: parse -> validate -> compute -> format."""

import re
from typing import Tuple

def parse_expression(expr: str) -> list:
    """Parse expression string into tokens."""
    # Tokenize: numbers, operators, parentheses
    tokens = re.findall(r'\d+\.?\d*|[+\-*/^()]', expr.replace(' ', ''))
    return tokens

def validate_tokens(tokens: list) -> bool:
    """Validate token sequence is a valid expression."""
    if not tokens:
        return False
    # Basic validation: balanced parens, no consecutive operators
    depth = 0
    expect_operand = True
    for t in tokens:
        if t == '(':
            depth += 1
            expect_operand = True
        elif t == ')':
            depth -= 1
            if depth < 0:
                return False
            expect_operand = False
        elif t in '+-*/^':
            if expect_operand:
                return False
            expect_operand = True

def format_result(result: float) -> str:
    """Format computation result."""
    if result == int(result):

    return f"{result:.4f}"

def evaluate(expr: str, flow_id: str = "") -> Tuple[str, str]:
    """
    Full evaluation pipeline: parse -> validate -> compute -> format.
    Returns (result_string, error_string).
    flow_id identifies the calling business flow for probe tracing.
    """
    # This is the shared path - all 3 flows call this function
    from codepoint import point_json

    # Stage 1: Parse
    point_json("cp-calc-parse", {
        "point_id": "cp-calc-parse",
        "flow_id": flow_id,
        "expr": expr,
    })
    tokens = parse_expression(expr)
    if not tokens:
        return "", "empty expression"

    # Stage 2: Validate
    point_json("cp-calc-validate", {
        "point_id": "cp-calc-validate",
        "flow_id": flow_id,
    })
    if not validate_tokens(tokens):
        return "", "invalid expression"

    # Stage 3: Compute
    point_json("cp-calc-compute", {
        "point_id": "cp-calc-compute",
        "flow_id": flow_id,
        "expr": expr,
    })
    result = compute(tokens)

    # Stage 4: Format
    point_json("cp-calc-format", {
        "point_id": "cp-calc-format",
        "flow_id": flow_id,
        "result": result,
    })
    return format_result(result), ""
```

**Important:** The `evaluate` function accepts `flow_id` parameter. This is how Python propagates flow identity without context.Context.

```python
def process_expressions(expressions: str, flow_id: str = "flow-batch-process") -> list:
    """Process multiple expressions in batch."""
    from codepoint import point_json
    point_json("cp-batch-entry", {
        "point_id": "cp-batch-entry",
        "flow_id": flow_id,
    })
    results = []
    for expr in expressions.strip().split('\n'):
        expr = expr.strip()
        if expr:
            result, err = evaluate(expr, flow_id=flow_id)
            results.append({"expr": expr, "result": result, "error": err})
    return results
```

## 1.6 Create history store (history/store.py)

```python
class HistoryStore:
    def __init__(self):
        self._records = {}
        self._next_id = 1

    def add(self, expr: str, result: str) -> int:
        record_id = self._next_id
        self._records[record_id] = {"id": record_id, "expression": expr, "result": result}
        self._next_id += 1
        return record_id

    def get(self, record_id: int) -> dict:
        return self._records.get(record_id)

    def get_all(self) -> list:
        return list(self._records.values())
```

```python
import sys
from batch.processor import process_expressions

if __name__ == "__main__":
    if len(sys.argv) > 1:
        # Single expression mode
        expr = sys.argv[1]
        from calculator.core import evaluate
        result, err = evaluate(expr, flow_id="flow-cli-single")
        if err:
            print(f"Error: {err}")
        else:
            print(f"Result: {result}")
    else:
        # Read from stdin
        print("Enter expressions (one per line, Ctrl+D to end):")
        expressions = sys.stdin.read()
        results = process_expressions(expressions)
        for r in results:
            print(f"{r['expr']} = {r['result']}" if not r['error'] else f"{r['expr']}: Error: {r['error']}")
```
