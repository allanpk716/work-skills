```javascript
let testCount = 0;
let passCount = 0;
const pendingPromises = [];

function test(name, fn) {
  testCount++;
  const num = testCount;
  try {
    const result = fn();
    if (result && typeof result.then === 'function') {
      const p = result.then(() => {
        console.log(`[OK] Test ${num}: ${name}`);
        passCount++;
      }).catch(error => {
        console.error(`[FAIL] Test ${num}: ${name}`);
        console.error(`    ${error.message || error}`);
      });
      pendingPromises.push(p);
      return p;
    }
    console.log(`[OK] Test ${num}: ${name}`);
    passCount++;
  } catch (error) {
    console.error(`[FAIL] Test ${num}: ${name}`);
    console.error(`    ${error.message || error}`);
  }
}

// Summary
(async () => {
  await Promise.all(pendingPromises);
  console.log('\n=== Test Summary ===');
  console.log(`Total: ${testCount}`);
  console.log(`Passed: ${passCount}`);
  console.log(`Failed: ${testCount - passCount}`);
  if (passCount === testCount) {
    console.log('\nAll tests passed!');
    process.exit(0);
  } else {
    console.log('\nSome tests failed.');
    process.exit(1);
  }
})();
```

**Core design: Mock enquirer to test the integration flow WITHOUT triggering interactive prompts.**

The test strategy is:
1. Mock `enquirer` Confirm and Input classes by overriding their `run()` methods to return predetermined values
2. Call the actual `configurePushover()`, `configureGitUser()`, `configureGitSSH()` functions with mocked enquirer
3. Verify each function returns the correct `{status, details}` for each scenario

**Implementation approach:**

Use `Module._resolveFilename` or direct require cache manipulation to inject a mock enquirer. Specifically:
```javascript
// Before requiring configurators, register a mock enquirer
const Module = require('module');
const originalResolve = Module._resolveFilename;
Module._resolveFilename = function(request, parent) {
  if (request === 'enquirer') {
    return 'mock-enquirer';
  }
  return originalResolve.apply(this, arguments);
};
require.cache['mock-enquirer'] = {
  id: 'mock-enquirer',
  filename: 'mock-enquirer',
  loaded: true,
  exports: mockEnquirer
};
```

OR simpler: clear require cache for each configurator module and set up mocks on the enquirer Confirm/Input prototypes before each test scenario.

**Simplest approach (recommended): Mock enquirer at the prototype level before requiring configurators:**

```javascript
const enquirer = require('enquirer');
```

Then before each scenario, configure mock responses:
```javascript
// Set up mock responses queue
let confirmResponses = [];
let inputResponses = [];

enquirer.Confirm.prototype.run = async function() {
  return confirmResponses.shift();
};
enquirer.Input.prototype.run = async function() {
  return inputResponses.shift();
};
```

**IMPORTANT: After reading git-ssh.test.js and git-user.test.js, you will notice that git-ssh.test.js tests 3-4 DO directly call configureGitSSH() and they work without mocking. This is because git-ssh.js does NOT use enquirer Input in the "already configured" path -- it only uses Confirm in the "not configured" path. However, to be safe and consistent, mock enquirer for ALL integration tests.**
