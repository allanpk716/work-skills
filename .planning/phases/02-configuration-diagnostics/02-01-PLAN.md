```python
def check_notification_flags():
    """
    Check for project-level notification disable flags.

    Returns:
        dict: {'pushover_disabled': bool, 'windows_disabled': bool}
    """
    project_dir = Path.cwd()

    flags = {
        'pushover_disabled': (project_dir / '.no-pushover').is_file(),
        'windows_disabled': (project_dir / '.no-windows').is_file()
    }

    if flags['pushover_disabled']:
        logger.info("Pushover notifications disabled by .no-pushover file")

    if flags['windows_disabled']:
        logger.info("Windows notifications disabled by .no-windows file")

    return flags
```

2. Modify `main()` function to check flags before sending notifications (around line 218):
```python
# Check project-level notification flags
flags = check_notification_flags()

# Send notifications in parallel
with ThreadPoolExecutor(max_workers=2) as executor:
    futures = {}

    # Only submit Pushover if not disabled
    if not flags['pushover_disabled']:
        futures[executor.submit(send_pushover_notification, project_name, summary)] = 'pushover'

    # Only submit Windows if not disabled
    if not flags['windows_disabled']:
        futures[executor.submit(send_windows_notification, project_name, summary)] = 'windows'

    # If both disabled, log and exit
    if not futures:
        logger.info("All notifications disabled by project flags")
        return 0
```

**Critical:**
- Use `Path.is_file()` NOT `Path.exists()` (handles directories correctly)
- Check in `Path.cwd()` (current working directory, which is the project root)
- Log when notifications are disabled (for debugging)
- Skip ThreadPoolExecutor submission entirely if disabled (saves resources)

```python
#!/usr/bin/env python3
"""Test script for Phase 2 configuration and control features."""

import os
import sys
import tempfile
from pathlib import Path

# Add parent directory to path for imports
sys.path.insert(0, str(Path(__file__).parent.parent / '.claude' / 'hooks'))

def test_environment_variables():
    """Test CONF-01 and CONF-02: Environment variable configuration."""
    print("\n[TEST 1] Environment Variable Configuration")

    # Save current values
    old_token = os.environ.get('PUSHOVER_TOKEN')
    old_user = os.environ.get('PUSHOVER_USER')

    try:
        # Test 1a: Missing credentials should not crash
        if 'PUSHOVER_TOKEN' in os.environ:
            del os.environ['PUSHOVER_TOKEN']
        if 'PUSHOVER_USER' in os.environ:
            del os.environ['PUSHOVER_USER']

        import notify
        # Should handle missing credentials gracefully
        result = notify.send_pushover_notification("Test", "Test")
        assert result == False, "Should return False when credentials missing"
        print("  ✓ Missing credentials handled gracefully")

        # Test 1b: Valid credentials should work
        os.environ['PUSHOVER_TOKEN'] = 'test_token_12345'
        os.environ['PUSHOVER_USER'] = 'test_user_12345'

        # Note: This will fail at API level, but should not crash
        # We're testing the configuration reading, not the API call
        print("  ✓ Credential reading works (API call may fail)")

    finally:
        # Restore original values
        if old_token:
            os.environ['PUSHOVER_TOKEN'] = old_token
        if old_user:
            os.environ['PUSHOVER_USER'] = old_user

def test_project_control_flags():
    """Test CONF-03 and CONF-04: Project-level notification control."""
    print("\n[TEST 2] Project-Level Control Flags")

    original_dir = os.getcwd()

    try:
        with tempfile.TemporaryDirectory() as tmpdir:
            os.chdir(tmpdir)

            import notify

            # Test 2a: .no-pushover file detection
            Path('.no-pushover').touch()
            flags = notify.check_notification_flags()
            assert flags['pushover_disabled'] == True, ".no-pushover should disable Pushover"
            assert flags['windows_disabled'] == False, "Windows should still be enabled"
            print("  ✓ .no-pushover file detected correctly")

            # Test 2b: .no-windows file detection
            Path('.no-pushover').unlink()
            Path('.no-windows').touch()
            flags = notify.check_notification_flags()
            assert flags['pushover_disabled'] == False, "Pushover should still be enabled"
            assert flags['windows_disabled'] == True, ".no-windows should disable Windows"
            print("  ✓ .no-windows file detected correctly")

            # Test 2c: Both disabled
            Path('.no-pushover').touch()
            flags = notify.check_notification_flags()
            assert flags['pushover_disabled'] == True
            assert flags['windows_disabled'] == True
            print("  ✓ Both notifications can be disabled")

            # Test 2d: No flags (all enabled)
            Path('.no-pushover').unlink()
            Path('.no-windows').unlink()
            flags = notify.check_notification_flags()
            assert flags['pushover_disabled'] == False
            assert flags['windows_disabled'] == False
            print("  ✓ No flags means all notifications enabled")

    finally:
        os.chdir(original_dir)

if __name__ == '__main__':
    print("=" * 60)
    print("Phase 2 - Configuration & Control Tests")
    print("=" * 60)

    test_environment_variables()
    test_project_control_flags()

    print("\n" + "=" * 60)
    print("All tests passed!")
    print("=" * 60)
```
